use chomp::{token, take};

    token(b'f');
    take(2)
});

use chomp::{token, many, take};
use chomp::buffer::{IntoStream, Stream};

let i = b"foofoo";

let r = i.into_stream().parse(parser!{many(parser!{
    token(b'f');
    take(2)
})});

assert_eq!(r, Ok(vec![b"oo" as &[u8], b"oo" as &[u8]]));
